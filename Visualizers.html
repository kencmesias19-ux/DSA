Stack (Infix to Postfix)

// Python Stack Example
stack = []
def push(val):
    stack.append(val)
def pop():
    if stack:
        return stack.pop()
push(10)
push(20)
print(pop())

Tree Traversals

// Python Binary Tree Example
class Node:
    def __init__(self, data):
        self.left = None
        self.right = None
        self.data = data

def inorder(root):
    if root:
        inorder(root.left)
        print(root.data, end=" ")
        inorder(root.right)

root = Node(1)
root.left = Node(2)
root.right = Node(3)
inorder(root)
